The Complete Guide to LLM Cost Optimization for AI Agents

March 5, 2024 · 8 min read
Marcus Johnson
Solutions Architect at AgentOps

Running AI agents in production can get expensive fast. We've seen teams spend $100,000+ per month on LLM API calls without realizing how much they could save. This guide shares the strategies that have helped our customers reduce costs by 50-80%.

Understanding LLM Costs​

Before optimizing, you need to understand where your money goes:

Most teams are surprised to learn that 15-20% of their costs come from redundant or unnecessary calls. That's the low-hanging fruit.

The Cost Optimization Framework​

We use a three-tier approach to cost optimization:

Tier 1: Quick Wins​

1. Remove Redundant Calls​

The first step is identifying and eliminating duplicate API calls:

import agentops
from functools import lru_cache
import hashlib

class OptimizedAgent:
    def __init__(self):
        self.call_cache = {}
        agentops.init(api_key="your-api-key")

    def _create_cache_key(self, prompt: str, model: str) -> str:
        """Create a deterministic cache key."""
        content = f"{model}:{prompt}"
        return hashlib.sha256(content.encode()).hexdigest()

    async def call_llm(self, prompt: str, model: str = "gpt-4"):
        cache_key = self._create_cache_key(prompt, model)

        # Check cache first
        if cache_key in self.call_cache:
            agentops.record(ActionEvent(
                action_type="cache_hit",
                params={"cache_key": cache_key[:8]}
            ))
            return self.call_cache[cache_key]

        # Make the actual call
        response = await self.client.chat.completions.create(
            model=model,
            messages=[{"role": "user", "content": prompt}]
        )

        # Cache the result
        self.call_cache[cache_key] = response.choices[0].message.content
        return self.call_cache[cache_key]

Impact: 15-25% cost reduction from caching alone.

2. Fix Retry Logic​

Exponential backoff with jitter prevents costly retry storms:

import asyncio
import random

async def call_with_smart_retry(func, max_retries=3, base_delay=1.0):
    """Smart retry with exponential backoff and jitter."""
    for attempt in range(max_retries):
        try:
            return await func()
        except RateLimitError:
            if attempt == max_retries - 1:
                raise

            # Exponential backoff with jitter
            delay = base_delay * (2 ** attempt) + random.uniform(0, 1)

            agentops.record(ActionEvent(
                action_type="rate_limit_retry",
                params={
                    "attempt": attempt + 1,
                    "delay": delay
                }
            ))

            await asyncio.sleep(delay)
        except Exception as e:
            # Don't retry on non-retryable errors
            agentops.record(ActionEvent(
                action_type="non_retryable_error",
                params={"error": str(e)}
            ))
            raise

3. Implement Basic Caching​

Use TTL-based caching for semi-dynamic content:

from datetime import datetime, timedelta
from dataclasses import dataclass

@dataclass
class CacheEntry:
    value: str
    expires_at: datetime
    hit_count: int = 0

class TTLCache:
    def __init__(self, default_ttl: timedelta = timedelta(hours=1)):
        self.cache: dict[str, CacheEntry] = {}
        self.default_ttl = default_ttl

    def get(self, key: str) -> str | None:
        if key in self.cache:
            entry = self.cache[key]
            if datetime.now() < entry.expires_at:
                entry.hit_count += 1
                return entry.value
            else:
                del self.cache[key]
        return None

    def set(self, key: str, value: str, ttl: timedelta = None):
        self.cache[key] = CacheEntry(
            value=value,
            expires_at=datetime.now() + (ttl or self.default_ttl)
        )

Tier 2: Architecture Optimization​

4. Model Selection Strategy​

Not every task needs GPT-4. Use the right model for the job:

Implement automatic model selection:

class ModelRouter:
    """Routes tasks to the most cost-effective model."""

    COMPLEXITY_MODELS = {
        "simple": "gpt-3.5-turbo",
        "medium": "gpt-4-turbo-preview",
        "complex": "gpt-4",
        "code": "claude-3-sonnet-20240229"
    }

    def __init__(self):
        self.classifier = self._load_classifier()

    def route(self, task: str) -> str:
        """Determine the best model for a task."""
        complexity = self._classify_complexity(task)

        agentops.record(ActionEvent(
            action_type="model_routed",
            params={
                "complexity": complexity,
                "model": self.COMPLEXITY_MODELS[complexity]
            }
        ))

        return self.COMPLEXITY_MODELS[complexity]

    def _classify_complexity(self, task: str) -> str:
        """Classify task complexity using a lightweight model."""
        # Use a simple heuristic or lightweight classifier
        if any(kw in task.lower() for kw in ["list", "format", "convert"]):
            return "simple"
        elif any(kw in task.lower() for kw in ["code", "function", "implement"]):
            return "code"
        elif any(kw in task.lower() for kw in ["analyze", "compare", "explain"]):
            return "medium"
        else:
            return "complex"

5. Prompt Optimization​

Shorter prompts = lower costs. But don't sacrifice quality:

class PromptOptimizer:
    """Optimizes prompts for cost without sacrificing quality."""

    def optimize(self, prompt: str) -> str:
        """Apply optimization techniques to a prompt."""
        optimized = prompt

        # Remove redundant whitespace
        optimized = " ".join(optimized.split())

        # Use abbreviations for common phrases
        replacements = {
            "Please provide": "Provide",
            "I would like you to": "",
            "Can you please": "",
            "It would be great if you could": "",
        }

        for old, new in replacements.items():
            optimized = optimized.replace(old, new)

        # Calculate savings
        original_tokens = len(prompt.split()) * 1.3  # Rough estimate
        optimized_tokens = len(optimized.split()) * 1.3

        agentops.record(ActionEvent(
            action_type="prompt_optimized",
            params={
                "original_tokens": int(original_tokens),
                "optimized_tokens": int(optimized_tokens),
                "savings_percent": round((1 - optimized_tokens/original_tokens) * 100, 1)
            }
        ))

        return optimized.strip()

6. Response Streaming​

Stream responses to fail fast and reduce wasted tokens:

async def stream_with_early_stop(prompt: str, stop_conditions: list[str]):
    """Stream response and stop early if conditions are met."""
    collected_text = ""
    tokens_saved = 0

    async for chunk in client.chat.completions.create(
        model="gpt-4",
        messages=[{"role": "user", "content": prompt}],
        stream=True
    ):
        content = chunk.choices[0].delta.content or ""
        collected_text += content

        # Check stop conditions
        for condition in stop_conditions:
            if condition in collected_text:
                agentops.record(ActionEvent(
                    action_type="early_stop",
                    params={
                        "condition": condition,
                        "tokens_collected": len(collected_text.split())
                    }
                ))
                return collected_text

    return collected_text

Tier 3: Advanced Optimization​

7. Semantic Caching​

Cache based on meaning, not exact matches:

import numpy as np
from openai import OpenAI

class SemanticCache:
    """Cache responses based on semantic similarity."""

    def __init__(self, similarity_threshold: float = 0.95):
        self.threshold = similarity_threshold
        self.embeddings: list[np.ndarray] = []
        self.responses: list[str] = []
        self.queries: list[str] = []

    def _get_embedding(self, text: str) -> np.ndarray:
        """Get embedding for text."""
        response = OpenAI().embeddings.create(
            model="text-embedding-3-small",
            input=text
        )
        return np.array(response.data[0].embedding)

    def _cosine_similarity(self, a: np.ndarray, b: np.ndarray) -> float:
        """Calculate cosine similarity between two vectors."""
        return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))

    def get(self, query: str) -> str | None:
        """Find semantically similar cached response."""
        query_embedding = self._get_embedding(query)

        for i, cached_embedding in enumerate(self.embeddings):
            similarity = self._cosine_similarity(query_embedding, cached_embedding)

            if similarity >= self.threshold:
                agentops.record(ActionEvent(
                    action_type="semantic_cache_hit",
                    params={
                        "similarity": round(similarity, 3),
                        "original_query": self.queries[i][:50]
                    }
                ))
                return self.responses[i]

        return None

    def set(self, query: str, response: str):
        """Add query-response pair to cache."""
        embedding = self._get_embedding(query)
        self.embeddings.append(embedding)
        self.responses.append(response)
        self.queries.append(query)

8. Cost Budgets and Alerts​

Prevent runaway costs with hard limits:

from dataclasses import dataclass
from datetime import datetime

@dataclass
class CostBudget:
    daily_limit: float
    monthly_limit: float
    alert_threshold: float = 0.8

class CostGuard:
    """Enforce cost budgets and send alerts."""

    def __init__(self, budget: CostBudget):
        self.budget = budget
        self.daily_spend = 0.0
        self.monthly_spend = 0.0
        self.last_reset = datetime.now()

    def check_and_record(self, cost: float) -> bool:
        """Check if cost is within budget, record spend."""
        self._maybe_reset()

        # Check daily limit
        if self.daily_spend + cost > self.budget.daily_limit:
            agentops.record(ActionEvent(
                action_type="budget_exceeded",
                params={
                    "type": "daily",
                    "limit": self.budget.daily_limit,
                    "current": self.daily_spend
                }
            ))
            return False

        # Record the spend
        self.daily_spend += cost
        self.monthly_spend += cost

        # Check alert thresholds
        if self.daily_spend > self.budget.daily_limit * self.budget.alert_threshold:
            self._send_alert("daily", self.daily_spend, self.budget.daily_limit)

        return True

    def _send_alert(self, period: str, current: float, limit: float):
        """Send cost alert."""
        agentops.record(ActionEvent(
            action_type="cost_alert",
            params={
                "period": period,
                "current_spend": current,
                "limit": limit,
                "percent_used": round(current / limit * 100, 1)
            }
        ))
