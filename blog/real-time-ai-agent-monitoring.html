Real-Time Monitoring for AI Agents: A Complete Implementation Guide

September 18, 2024 · 9 min read
Chris Anderson
Platform Engineer at AgentOps

When your AI agent is handling thousands of requests per minute, you need to know the moment something goes wrong. This guide shows you how to build a comprehensive real-time monitoring system for AI agents.

The Monitoring Stack​

A complete monitoring system has multiple layers:

Essential Metrics to Track​

1. Operational Metrics​

from dataclasses import dataclass, field
from collections import deque
import time
import agentops

@dataclass
class OperationalMetrics:
    """Track operational health metrics."""

    # Rolling windows for different time periods
    requests_1m: deque = field(default_factory=lambda: deque(maxlen=60))
    requests_5m: deque = field(default_factory=lambda: deque(maxlen=300))
    errors_1m: deque = field(default_factory=lambda: deque(maxlen=60))
    latencies_1m: deque = field(default_factory=lambda: deque(maxlen=1000))

    def record_request(self, success: bool, latency_ms: float):
        """Record a request outcome."""
        now = time.time()

        self.requests_1m.append(now)
        self.requests_5m.append(now)
        self.latencies_1m.append(latency_ms)

        if not success:
            self.errors_1m.append(now)

        # Send to AgentOps for persistence
        agentops.record(ActionEvent(
            action_type="request_recorded",
            params={
                "success": success,
                "latency_ms": latency_ms
            }
        ))

    def get_health_score(self) -> dict:
        """Calculate current health score."""
        now = time.time()
        cutoff_1m = now - 60

        # Calculate metrics
        request_count = sum(1 for t in self.requests_1m if t > cutoff_1m)
        error_count = sum(1 for t in self.errors_1m if t > cutoff_1m)
        error_rate = error_count / max(request_count, 1)

        latencies = list(self.latencies_1m)
        p50 = np.percentile(latencies, 50) if latencies else 0
        p99 = np.percentile(latencies, 99) if latencies else 0

        # Calculate health score (0-100)
        error_penalty = min(error_rate * 100, 50)  # Max 50 points lost
        latency_penalty = min((p99 / 5000) * 30, 30)  # Max 30 points for latency
        health_score = 100 - error_penalty - latency_penalty

        return {
            "health_score": max(health_score, 0),
            "requests_per_minute": request_count,
            "error_rate": error_rate,
            "latency_p50": p50,
            "latency_p99": p99,
            "status": self._get_status(health_score)
        }

    def _get_status(self, score: float) -> str:
        if score >= 90:
            return "healthy"
        elif score >= 70:
            return "warning"
        else:
            return "critical"

2. LLM-Specific Metrics​

Track metrics unique to LLM-powered agents:

class LLMMetricsCollector:
    """Collect LLM-specific metrics."""

    def __init__(self):
        self.token_usage = {"input": 0, "output": 0}
        self.costs = {"total": 0.0}
        self.model_calls = defaultdict(int)
        self.cache_stats = {"hits": 0, "misses": 0}

    def record_llm_call(
        self,
        model: str,
        input_tokens: int,
        output_tokens: int,
        cost: float,
        cached: bool = False
    ):
        """Record an LLM API call."""
        self.model_calls[model] += 1
        self.token_usage["input"] += input_tokens
        self.token_usage["output"] += output_tokens
        self.costs["total"] += cost

        if cached:
            self.cache_stats["hits"] += 1
        else:
            self.cache_stats["misses"] += 1

        agentops.record(ActionEvent(
            action_type="llm_call",
            params={
                "model": model,
                "input_tokens": input_tokens,
                "output_tokens": output_tokens,
                "cost": cost,
                "cached": cached
            }
        ))

    def get_summary(self) -> dict:
        """Get current LLM usage summary."""
        total_cache_requests = self.cache_stats["hits"] + self.cache_stats["misses"]
        cache_rate = self.cache_stats["hits"] / max(total_cache_requests, 1)

        return {
            "total_tokens": sum(self.token_usage.values()),
            "input_tokens": self.token_usage["input"],
            "output_tokens": self.token_usage["output"],
            "total_cost": self.costs["total"],
            "calls_by_model": dict(self.model_calls),
            "cache_hit_rate": cache_rate
        }

3. Agent Behavior Metrics​

Track how agents make decisions:

class AgentBehaviorTracker:
    """Track agent decision-making patterns."""

    def __init__(self):
        self.tool_usage = defaultdict(int)
        self.decision_paths = defaultdict(int)
        self.reasoning_lengths = []

    def record_tool_use(self, tool_name: str, success: bool, latency_ms: float):
        """Record a tool invocation."""
        self.tool_usage[tool_name] += 1

        agentops.record(ActionEvent(
            action_type="tool_used",
            params={
                "tool": tool_name,
                "success": success,
                "latency_ms": latency_ms
            }
        ))

    def record_decision_path(self, path: list[str]):
        """Record the path of decisions/tools used."""
        path_key = " -> ".join(path)
        self.decision_paths[path_key] += 1

        agentops.record(ActionEvent(
            action_type="decision_path",
            params={
                "path": path,
                "path_length": len(path)
            }
        ))

    def get_insights(self) -> dict:
        """Get behavioral insights."""
        total_tool_calls = sum(self.tool_usage.values())

        return {
            "tool_distribution": {
                tool: count / total_tool_calls
                for tool, count in self.tool_usage.items()
            },
            "most_common_paths": sorted(
                self.decision_paths.items(),
                key=lambda x: x[1],
                reverse=True
            )[:10],
            "avg_path_length": np.mean([
                len(path.split(" -> "))
                for path in self.decision_paths.keys()
            ])
        }

Alert Rules​

Define rules for when to alert:

from enum import Enum
from dataclasses import dataclass

class AlertSeverity(Enum):
    INFO = "info"
    WARNING = "warning"
    ERROR = "error"
    CRITICAL = "critical"

@dataclass
class AlertRule:
    name: str
    condition: str  # Python expression
    severity: AlertSeverity
    cooldown_seconds: int = 300  # Don't re-alert for 5 minutes
    message_template: str = ""

# Define alert rules
ALERT_RULES = [
    AlertRule(
        name="high_error_rate",
        condition="metrics['error_rate'] > 0.1",
        severity=AlertSeverity.ERROR,
        message_template="Error rate is {error_rate:.1%}, exceeds 10% threshold"
    ),
    AlertRule(
        name="critical_error_rate",
        condition="metrics['error_rate'] > 0.25",
        severity=AlertSeverity.CRITICAL,
        cooldown_seconds=60,
        message_template="CRITICAL: Error rate is {error_rate:.1%}!"
    ),
    AlertRule(
        name="high_latency",
        condition="metrics['latency_p99'] > 5000",
        severity=AlertSeverity.WARNING,
        message_template="P99 latency is {latency_p99:.0f}ms, exceeds 5s threshold"
    ),
    AlertRule(
        name="cost_spike",
        condition="metrics['cost_rate'] > hourly_budget",

Alert Engine​

Process rules and send alerts:

import asyncio
from datetime import datetime, timedelta

class AlertEngine:
    """Process alert rules and send notifications."""

    def __init__(self, rules: list[AlertRule]):
        self.rules = rules
        self.last_alerts: dict[str, datetime] = {}
        self.alert_handlers = []

    def add_handler(self, handler):
        """Add an alert handler (Slack, PagerDuty, etc.)."""
        self.alert_handlers.append(handler)

    async def evaluate(self, metrics: dict):
        """Evaluate all rules against current metrics."""
        for rule in self.rules:
            try:
                # Evaluate condition
                triggered = eval(rule.condition, {"metrics": metrics})

                if triggered:
                    await self._maybe_alert(rule, metrics)

            except Exception as e:
                # Log rule evaluation errors but don't crash
                agentops.record(ActionEvent(
                    action_type="alert_rule_error",
                    params={
                        "rule": rule.name,
                        "error": str(e)
                    }
                ))

    async def _maybe_alert(self, rule: AlertRule, metrics: dict):
        """Send alert if not in cooldown."""
        now = datetime.now()
        last_alert = self.last_alerts.get(rule.name)

        if last_alert:
            cooldown_end = last_alert + timedelta(seconds=rule.cooldown_seconds)
            if now < cooldown_end:
                return  # Still in cooldown

        # Generate alert
        alert = {
            "rule": rule.name,
            "severity": rule.severity.value,
            "message": rule.message_template.format(**metrics),
            "timestamp": now.isoformat(),
            "metrics": metrics
        }

        # Record in AgentOps
        agentops.record(ActionEvent(
            action_type="alert_triggered",
            params=alert
        ))

        # Send to all handlers
        for handler in self.alert_handlers:
            await handler.send(alert)

        self.last_alerts[rule.name] = now


class SlackAlertHandler:
    """Send alerts to Slack."""

    def __init__(self, webhook_url: str):
        self.webhook_url = webhook_url

    async def send(self, alert: dict):
        """Send alert to Slack."""
        color = {
            "info": "#36a64f",
            "warning": "#FFA500",
            "error": "#FF6B6B",
            "critical": "#FF0000"
        }.get(alert["severity"], "#808080")

        payload = {
            "attachments": [{
                "color": color,
                "title": f"🚨 {alert['rule']}",
                "text": alert["message"],
                "fields": [
                    {"title": "Severity", "value": alert["severity"], "short": True},
                    {"title": "Time", "value": alert["timestamp"], "short": True}
                ]
            }]
        }

        async with aiohttp.ClientSession() as session:
            await session.post(self.webhook_url, json=payload)

Backend Implementation​

from fastapi import FastAPI, WebSocket
from fastapi.websockets import WebSocketDisconnect
import asyncio

app = FastAPI()

class DashboardServer:
    """Real-time dashboard WebSocket server."""

    def __init__(self):
        self.connections: list[WebSocket] = []
        self.metrics = OperationalMetrics()
        self.llm_metrics = LLMMetricsCollector()
        self.behavior = AgentBehaviorTracker()

    async def connect(self, websocket: WebSocket):
        """Handle new dashboard connection."""
        await websocket.accept()
        self.connections.append(websocket)

        # Send initial state
        await self._send_full_state(websocket)

    async def disconnect(self, websocket: WebSocket):
        """Handle dashboard disconnection."""
        self.connections.remove(websocket)

    async def broadcast_update(self, update: dict):
        """Send update to all connected dashboards."""
        disconnected = []

        for ws in self.connections:
            try:
                await ws.send_json(update)
            except:
                disconnected.append(ws)

        for ws in disconnected:
            self.connections.remove(ws)

    async def _send_full_state(self, websocket: WebSocket):
        """Send complete current state to new connection."""
        state = {
            "type": "full_state",
            "health": self.metrics.get_health_score(),
            "llm": self.llm_metrics.get_summary(),
            "behavior": self.behavior.get_insights()
        }
        await websocket.send_json(state)

    async def metrics_loop(self):
        """Continuously broadcast metrics updates."""
        while True:
            update = {
                "type": "metrics_update",
                "timestamp": time.time(),
                "health": self.metrics.get_health_score(),
                "llm": self.llm_metrics.get_summary()
            }
            await self.broadcast_update(update)
            await asyncio.sleep(1)  # Update every second


dashboard = DashboardServer()

@app.websocket("/ws/dashboard")
async def dashboard_endpoint(websocket: WebSocket):
    await dashboard.connect(websocket)
    try:
        while True:
            # Keep connection alive, handle any commands
            data = await websocket.receive_json()
            if data.get("command") == "refresh":
                await dashboard._send_full_state(websocket)
    except WebSocketDisconnect:
        await dashboard.disconnect(websocket)

Integrating with AgentOps​

AgentOps provides built-in monitoring that complements custom solutions:

import agentops
from agentops import track_agent, ActionEvent

# Initialize with monitoring options
agentops.init(
    api_key="your-api-key",
    default_tags=["production", "monitored"],
    auto_start_session=True
)

@track_agent(name="ProductionAgent")
class ProductionAgent:
    """Agent with full monitoring integration."""

    def __init__(self, dashboard: DashboardServer):
        self.dashboard = dashboard

    async def run(self, request: str) -> str:
        start_time = time.time()

        try:
            # Process request
            result = await self._process(request)

            # Record success
            latency = (time.time() - start_time) * 1000
            self.dashboard.metrics.record_request(True, latency)

            return result

        except Exception as e:
            # Record failure
            latency = (time.time() - start_time) * 1000
            self.dashboard.metrics.record_request(False, latency)

            agentops.record(ActionEvent(
                action_type="agent_error",
                params={
                    "error_type": type(e).__name__,
                    "error_message": str(e)
                }
            ))

            raise

Conclusion​

Effective real-time monitoring for AI agents requires:

  1. Multi-dimensional metrics - Track operations, LLM usage, and behavior
  2. Intelligent alerting - Rules with cooldowns and severity levels
  3. Real-time visibility - Dashboards that update instantly
  4. Integration - Connect with AgentOps for comprehensive tracking

With this monitoring stack, you'll catch issues before they become incidents and have the data to continuously improve your agents.
