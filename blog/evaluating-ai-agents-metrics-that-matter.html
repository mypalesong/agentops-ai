Evaluating AI Agents: The Metrics That Actually Matter

July 15, 2024 · 9 min read
Michael Torres
Head of AI Research at AgentOps

"Our agent has 95% accuracy!" Great, but is it actually useful? Traditional ML metrics don't capture the full picture of agent performance. Let's explore the metrics that truly matter.

Beyond Accuracy: The Agent Evaluation Framework​

AI agent evaluation requires a multi-dimensional approach:

Quality Metrics​

Task Completion Rate​

Not just "did it finish?" but "did it finish correctly?"

from dataclasses import dataclass
from enum import Enum
import agentops

class CompletionStatus(Enum):
    COMPLETE = "complete"
    PARTIAL = "partial"
    FAILED = "failed"
    TIMEOUT = "timeout"

@dataclass
class TaskResult:
    status: CompletionStatus
    output: str
    required_elements: list[str]
    found_elements: list[str]

class TaskEvaluator:
    """Evaluate task completion quality."""

    def evaluate(self, task: str, result: TaskResult) -> dict:
        # Calculate completion score
        if result.required_elements:
            completion_rate = len(result.found_elements) / len(result.required_elements)
        else:
            completion_rate = 1.0 if result.status == CompletionStatus.COMPLETE else 0.0

        evaluation = {
            "status": result.status.value,
            "completion_rate": completion_rate,
            "missing_elements": [
                e for e in result.required_elements
                if e not in result.found_elements
            ]
        }

        agentops.record(ActionEvent(
            action_type="task_evaluation",
            params=evaluation
        ))

        return evaluation

Output Quality Score​

Measure the quality of what the agent produces:

class OutputQualityEvaluator:
    """Evaluate the quality of agent outputs."""

    def __init__(self, evaluator_llm):
        self.evaluator = evaluator_llm

    async def evaluate(self, task: str, output: str, criteria: dict) -> dict:
        """
        Evaluate output quality using LLM-as-judge.

        criteria example:
        {
            "factual_accuracy": "Output contains only verifiable facts",
            "relevance": "Output directly addresses the task",
            "completeness": "Output covers all aspects of the task",
            "format": "Output follows the requested format"
        }
        """
        scores = {}

        for criterion, description in criteria.items():
            prompt = f"""
            Task: {task}
            Output: {output}

            Evaluate the output on this criterion:
            {criterion}: {description}

            Score from 1-5 (5 being best) and explain briefly.
            Format: SCORE: X
            REASON: ...
            """

            result = await self.evaluator.generate(prompt)
            score = self._extract_score(result)
            scores[criterion] = score

        # Calculate weighted average
        weights = {"factual_accuracy": 0.4, "relevance": 0.3,
                   "completeness": 0.2, "format": 0.1}
        weighted_score = sum(
            scores.get(k, 0) * v for k, v in weights.items()
        ) / sum(weights.values())

        evaluation = {
            "individual_scores": scores,
            "weighted_score": weighted_score,
            "max_score": 5.0
        }

        agentops.record(ActionEvent(
            action_type="quality_evaluation",
            params=evaluation
        ))

        return evaluation

Reasoning Quality​

Evaluate the agent's decision-making process:

class ReasoningEvaluator:
    """Evaluate the quality of agent reasoning."""

    def evaluate_reasoning_chain(self, steps: list[dict]) -> dict:
        """
        Evaluate a chain of reasoning steps.

        Each step should have:
        - thought: The agent's reasoning
        - action: What action was taken
        - result: The outcome
        """
        metrics = {
            "logical_consistency": self._check_consistency(steps),
            "efficiency": self._check_efficiency(steps),
            "grounding": self._check_grounding(steps),
            "adaptability": self._check_adaptability(steps)
        }

        agentops.record(ActionEvent(
            action_type="reasoning_evaluation",
            params=metrics
        ))

        return metrics

    def _check_consistency(self, steps: list[dict]) -> float:
        """Check if reasoning is logically consistent."""
        # Look for contradictions between steps
        contradictions = 0
        for i, step in enumerate(steps[1:], 1):
            if self._contradicts_previous(step, steps[:i]):
                contradictions += 1

        return 1.0 - (contradictions / max(len(steps) - 1, 1))

    def _check_efficiency(self, steps: list[dict]) -> float:
        """Check if the agent took unnecessary steps."""
        # Compare actual steps to estimated minimum
        estimated_minimum = self._estimate_minimum_steps(steps)
        efficiency = estimated_minimum / len(steps)
        return min(efficiency, 1.0)

    def _check_grounding(self, steps: list[dict]) -> float:
        """Check if decisions are based on evidence."""
        grounded_steps = sum(
            1 for step in steps
            if step.get("evidence") or step.get("source")
        )
        return grounded_steps / len(steps)

    def _check_adaptability(self, steps: list[dict]) -> float:
        """Check if agent adapted to new information."""
        adaptations = sum(
            1 for i, step in enumerate(steps[1:], 1)
            if self._shows_adaptation(step, steps[i-1])
        )
        opportunities = sum(
            1 for step in steps
            if step.get("new_information")
        )
        return adaptations / max(opportunities, 1)

Efficiency Metrics​

Latency Distribution​

Track not just average latency, but the full distribution:

import numpy as np
from collections import defaultdict

class LatencyTracker:
    """Track and analyze agent latency."""

    def __init__(self):
        self.latencies = defaultdict(list)

    def record(self, operation: str, latency_ms: float):
        """Record a latency measurement."""
        self.latencies[operation].append(latency_ms)

        agentops.record(ActionEvent(
            action_type="latency_recorded",
            params={
                "operation": operation,
                "latency_ms": latency_ms
            }
        ))

    def get_percentiles(self, operation: str) -> dict:
        """Get latency percentiles for an operation."""
        data = self.latencies[operation]
        if not data:
            return {}

        return {
            "p50": np.percentile(data, 50),
            "p75": np.percentile(data, 75),
            "p90": np.percentile(data, 90),
            "p95": np.percentile(data, 95),
            "p99": np.percentile(data, 99),
            "mean": np.mean(data),
            "std": np.std(data)
        }

    def detect_anomalies(self, operation: str, threshold_std: float = 2.0) -> list:
        """Detect anomalous latencies."""
        data = self.latencies[operation]
        mean = np.mean(data)
        std = np.std(data)

        anomalies = [
            (i, lat) for i, lat in enumerate(data)
            if abs(lat - mean) > threshold_std * std
        ]

        return anomalies

Token Efficiency​

Measure how efficiently the agent uses tokens:

class TokenEfficiencyAnalyzer:
    """Analyze token usage efficiency."""

    def analyze(self, session_data: dict) -> dict:
        """
        Analyze token efficiency for a session.

        session_data should contain:
        - llm_calls: List of LLM call data
        - task_complexity: Estimated task complexity
        - output_quality: Quality score of final output
        """
        total_tokens = sum(
            call["input_tokens"] + call["output_tokens"]
            for call in session_data["llm_calls"]
        )

        # Calculate efficiency metrics
        metrics = {
            "total_tokens": total_tokens,
            "tokens_per_quality_point": total_tokens / max(session_data["output_quality"], 0.1),
            "input_output_ratio": self._calc_io_ratio(session_data["llm_calls"]),
            "redundancy_score": self._calc_redundancy(session_data["llm_calls"]),
            "estimated_optimal_tokens": self._estimate_optimal(session_data),
            "efficiency_percentage": None  # Calculated below
        }

        metrics["efficiency_percentage"] = (
            metrics["estimated_optimal_tokens"] / total_tokens * 100
        )

        agentops.record(ActionEvent(
            action_type="token_efficiency",
            params=metrics
        ))

        return metrics

    def _calc_redundancy(self, llm_calls: list) -> float:
        """Calculate how much content is repeated across calls."""
        # Simplified: check for repeated prompt content
        prompts = [call["prompt"] for call in llm_calls]
        unique_content = set()
        total_content = 0

        for prompt in prompts:
            words = prompt.split()
            unique_content.update(words)
            total_content += len(words)

        return 1 - (len(unique_content) / max(total_content, 1))

Error Analysis​

Categorize and track error patterns:

class ErrorAnalyzer:
    """Analyze agent error patterns."""

    ERROR_CATEGORIES = {
        "hallucination": ["incorrect fact", "made up", "fabricated"],
        "incomplete": ["missing", "partial", "truncated"],
        "off_topic": ["irrelevant", "unrelated", "tangent"],
        "format_error": ["wrong format", "invalid json", "parse error"],
        "timeout": ["timed out", "too slow", "deadline exceeded"],
        "tool_failure": ["tool error", "api failed", "external service"]
    }

    def __init__(self):
        self.error_counts = defaultdict(int)
        self.error_examples = defaultdict(list)

    def categorize_error(self, error_message: str, context: dict) -> str:
        """Categorize an error into predefined categories."""
        error_lower = error_message.lower()

        for category, keywords in self.ERROR_CATEGORIES.items():
            if any(kw in error_lower for kw in keywords):
                self.error_counts[category] += 1
                self.error_examples[category].append({
                    "message": error_message,
                    "context": context
                })

                agentops.record(ActionEvent(
                    action_type="error_categorized",
                    params={
                        "category": category,
                        "message": error_message[:200]
                    }
                ))

                return category

        self.error_counts["unknown"] += 1
        return "unknown"

    def get_error_report(self) -> dict:
        """Generate error analysis report."""
        total_errors = sum(self.error_counts.values())

        return {
            "total_errors": total_errors,
            "by_category": dict(self.error_counts),
            "percentages": {
                cat: count / total_errors * 100
                for cat, count in self.error_counts.items()
            },
            "top_issues": sorted(
                self.error_counts.items(),
                key=lambda x: x[1],
                reverse=True
            )[:5]
        }

ROI Calculator​

Calculate the business value of your agent:

class AgentROICalculator:
    """Calculate ROI for AI agent deployment."""

    def calculate(
        self,
        agent_costs: dict,
        manual_baseline: dict,
        volume: int
    ) -> dict:
        """
        Calculate ROI comparing agent to manual process.

        agent_costs: {
            "llm_cost_per_task": 0.05,
            "infrastructure_monthly": 500,
            "maintenance_hours_monthly": 10,
            "hourly_rate": 50
        }

        manual_baseline: {
            "time_per_task_minutes": 15,
            "hourly_rate": 30,
            "error_rate": 0.05,
            "error_cost": 100
        }
        """
        # Agent costs
        agent_llm_cost = agent_costs["llm_cost_per_task"] * volume
        agent_infra_cost = agent_costs["infrastructure_monthly"]
        agent_maintenance = (
            agent_costs["maintenance_hours_monthly"] *
            agent_costs["hourly_rate"]
        )
        total_agent_cost = agent_llm_cost + agent_infra_cost + agent_maintenance

        # Manual costs
        manual_time_hours = (manual_baseline["time_per_task_minutes"] * volume) / 60
        manual_labor_cost = manual_time_hours * manual_baseline["hourly_rate"]
        manual_error_cost = (
            volume * manual_baseline["error_rate"] * manual_baseline["error_cost"]
        )
        total_manual_cost = manual_labor_cost + manual_error_cost

        # ROI calculation
        savings = total_manual_cost - total_agent_cost
        roi_percentage = (savings / total_agent_cost) * 100

        result = {
            "agent_cost": total_agent_cost,
            "manual_cost": total_manual_cost,
            "monthly_savings": savings,
            "roi_percentage": roi_percentage,
            "breakeven_volume": self._calc_breakeven(agent_costs, manual_baseline),
            "time_saved_hours": manual_time_hours
        }

        agentops.record(ActionEvent(
            action_type="roi_calculated",
            params=result
        ))

        return result

Building an Evaluation Pipeline​

Combine all metrics into a comprehensive evaluation:

class AgentEvaluationPipeline:
    """Complete evaluation pipeline for AI agents."""

    def __init__(self, thresholds: dict):
        self.thresholds = thresholds
        self.quality_evaluator = OutputQualityEvaluator()
        self.latency_tracker = LatencyTracker()
        self.error_analyzer = ErrorAnalyzer()

    async def evaluate(self, agent, test_cases: list[dict]) -> dict:
        """Run full evaluation on an agent."""
        results = {
            "test_cases": len(test_cases),
            "passed": 0,
            "failed": 0,
            "scores": []
        }

        for test_case in test_cases:
            start = time.time()
            try:
                output = await agent.run(test_case["input"])
                latency = (time.time() - start) * 1000

                # Record latency
                self.latency_tracker.record("agent_run", latency)

                # Evaluate quality
                quality = await self.quality_evaluator.evaluate(
                    test_case["input"],
                    output,
                    test_case.get("criteria", {})
                )

                # Check expected output if provided
                if "expected" in test_case:
                    match_score = self._compare_outputs(
                        output, test_case["expected"]
                    )
                    quality["match_score"] = match_score

                results["scores"].append(quality)

                if quality["weighted_score"] >= self.thresholds["quality"]:
                    results["passed"] += 1
                else:
                    results["failed"] += 1

            except Exception as e:
                results["failed"] += 1
                self.error_analyzer.categorize_error(str(e), test_case)

        # Aggregate results
        results["summary"] = self._summarize(results)
        results["latency"] = self.latency_tracker.get_percentiles("agent_run")
        results["errors"] = self.error_analyzer.get_error_report()
        results["recommendation"] = self._make_recommendation(results)

        return results

    def _make_recommendation(self, results: dict) -> str:
        """Generate deployment recommendation."""
        success_rate = results["passed"] / results["test_cases"]
        avg_quality = np.mean([s["weighted_score"] for s in results["scores"]])
        p95_latency = results["latency"].get("p95", float("inf"))

        if (success_rate >= self.thresholds["success_rate"] and
            avg_quality >= self.thresholds["quality"] and
            p95_latency <= self.thresholds["latency_p95"]):
            return "READY_FOR_PRODUCTION"
        elif success_rate >= 0.8:
            return "NEEDS_IMPROVEMENT"
        else:
            return "NOT_READY"

Conclusion​

Effective agent evaluation goes far beyond simple accuracy metrics. By measuring quality, efficiency, reliability, and business impact, you get a complete picture of agent performance.

Key takeaways:

  1. Quality is multi-dimensional - Measure completion, output quality, and reasoning
  2. Efficiency matters - Track latency distributions and token usage
  3. Reliability requires tracking - Categorize errors and monitor trends
  4. Business metrics close the loop - Calculate actual ROI
