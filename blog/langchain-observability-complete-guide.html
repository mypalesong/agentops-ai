LangChain + AgentOps: Complete Observability Guide

April 12, 2024 · 8 min read
Emily Zhang
Developer Advocate at AgentOps

LangChain is one of the most popular frameworks for building AI agents, but debugging LangChain applications can be challenging. This guide shows you how to add comprehensive observability to your LangChain agents using AgentOps.

Why Observability Matters for LangChain​

LangChain applications involve multiple components working together:

Without observability, you're flying blind. You can't see:

  • Which tools are being called and why
  • How much each LLM call costs
  • Where bottlenecks occur in the chain
  • Why the agent made certain decisions

Quick Setup​

Getting started takes just two lines of code:

import agentops
agentops.init(api_key="your-api-key")

# Your existing LangChain code works unchanged!
from langchain.agents import create_openai_functions_agent, AgentExecutor
from langchain_openai import ChatOpenAI
from langchain.tools import tool

@tool
def search_web(query: str) -> str:
    """Search the web for information."""
    # Your search implementation
    return f"Results for: {query}"

llm = ChatOpenAI(model="gpt-4")
agent = create_openai_functions_agent(llm, [search_web], prompt)
executor = AgentExecutor(agent=agent, tools=[search_web])

# AgentOps automatically tracks everything!
result = executor.invoke({"input": "What's the weather in Tokyo?"})

That's it! AgentOps automatically instruments:

  • All LLM calls with prompts and responses
  • Tool invocations with inputs and outputs
  • Agent reasoning steps
  • Token usage and costs

Tool Executions​

Tool calls are tracked with timing and results:

from langchain.tools import tool
from agentops import ActionEvent

@tool
def database_query(query: str) -> str:
    """Execute a database query and return results."""
    # AgentOps automatically tracks this tool execution
    # But you can add custom events for more detail:

    agentops.record(ActionEvent(
        action_type="database_query",
        params={
            "query": query,
            "table": extract_table_name(query)
        }
    ))

    results = execute_query(query)

    agentops.record(ActionEvent(
        action_type="database_result",
        params={
            "row_count": len(results),
            "columns": list(results.columns)
        }
    ))

    return results.to_string()

Advanced Integration Patterns​

Custom Callbacks​

For fine-grained control, use custom callbacks:

from langchain.callbacks.base import BaseCallbackHandler
from agentops import ActionEvent

class AgentOpsCallback(BaseCallbackHandler):
    """Custom callback for detailed AgentOps integration."""

    def on_chain_start(self, serialized, inputs, **kwargs):
        agentops.record(ActionEvent(
            action_type="chain_start",
            params={
                "chain_type": serialized.get("name", "unknown"),
                "inputs": str(inputs)[:500]  # Truncate for storage
            }
        ))

    def on_chain_end(self, outputs, **kwargs):
        agentops.record(ActionEvent(
            action_type="chain_end",
            params={"outputs": str(outputs)[:500]}
        ))

    def on_agent_action(self, action, **kwargs):
        agentops.record(ActionEvent(
            action_type="agent_action",
            params={
                "tool": action.tool,
                "tool_input": str(action.tool_input)[:500],
                "log": action.log[:200]
            }
        ))

    def on_tool_error(self, error, **kwargs):
        agentops.record(ActionEvent(
            action_type="tool_error",
            params={
                "error_type": type(error).__name__,
                "error_message": str(error)
            }
        ))

# Use the callback
executor = AgentExecutor(
    agent=agent,
    tools=tools,
    callbacks=[AgentOpsCallback()]
)

Tracking RAG Pipelines​

For Retrieval-Augmented Generation:

from langchain.vectorstores import Chroma
from langchain.embeddings import OpenAIEmbeddings
from langchain.chains import RetrievalQA

# Initialize with AgentOps tracking
agentops.init(api_key="your-api-key")

# Create your RAG pipeline
embeddings = OpenAIEmbeddings()
vectorstore = Chroma(embedding_function=embeddings)

# Track retrieval metrics
class TrackedRetriever:
    def __init__(self, retriever):
        self.retriever = retriever

    def get_relevant_documents(self, query: str):
        start_time = time.time()

        docs = self.retriever.get_relevant_documents(query)

        agentops.record(ActionEvent(
            action_type="retrieval",
            params={
                "query": query,
                "num_docs_retrieved": len(docs),
                "latency_ms": (time.time() - start_time) * 1000,
                "doc_sources": [doc.metadata.get("source") for doc in docs]
            }
        ))

        return docs

# Use in your chain
qa_chain = RetrievalQA.from_chain_type(
    llm=ChatOpenAI(),
    retriever=TrackedRetriever(vectorstore.as_retriever())
)

Multi-Chain Workflows​

Track complex workflows with multiple chains:

from langchain.chains import SequentialChain

# Create multiple chains
research_chain = LLMChain(llm=llm, prompt=research_prompt, output_key="research")
analysis_chain = LLMChain(llm=llm, prompt=analysis_prompt, output_key="analysis")
summary_chain = LLMChain(llm=llm, prompt=summary_prompt, output_key="summary")

# Combine into sequential chain
full_pipeline = SequentialChain(
    chains=[research_chain, analysis_chain, summary_chain],
    input_variables=["topic"],
    output_variables=["research", "analysis", "summary"]
)

# Track with custom session tags
with agentops.start_session(tags=["multi-chain", "research-pipeline"]):
    result = full_pipeline.invoke({"topic": "AI in healthcare"})

    # Add session-level metrics
    agentops.record(ActionEvent(
        action_type="pipeline_complete",
        params={
            "topic": "AI in healthcare",
            "research_length": len(result["research"]),
            "analysis_length": len(result["analysis"]),
            "summary_length": len(result["summary"])
        }
    ))

Debugging Common Issues​

Issue 1: Agent Loops​

When your agent gets stuck in a loop:

AgentOps shows you:

  • The exact sequence of tool calls
  • What inputs triggered the loop
  • How many iterations occurred

Issue 2: High Latency​

Identify bottlenecks in your chain:

# AgentOps automatically tracks timing, but you can add custom spans
with agentops.start_span("custom_processing"):
    # Your slow operation
    processed_data = expensive_operation(data)

# In the dashboard, you'll see timing breakdown:
# - LLM calls: 2.5s
# - Tool executions: 1.2s
# - Custom processing: 0.8s
# - Other: 0.1s

Issue 3: Unexpected Costs​

Track costs in real-time:

# Set up cost alerts
agentops.init(
    api_key="your-api-key",
    tags=["production"],
    # Get alerted if session cost exceeds $1
    max_session_cost=1.0
)

Real-World Example​

Here's a complete example of a production-ready LangChain agent with full observability:

import agentops
from langchain.agents import create_openai_functions_agent, AgentExecutor
from langchain_openai import ChatOpenAI
from langchain.tools import tool
from langchain.prompts import ChatPromptTemplate, MessagesPlaceholder

# Initialize AgentOps
agentops.init(
    api_key="your-api-key",
    default_tags=["production", "customer-support"]
)

# Define tools with clear descriptions
@tool
def search_knowledge_base(query: str) -> str:
    """Search our knowledge base for relevant articles."""
    results = kb_client.search(query, limit=3)
    return "\n".join([r.content for r in results])

@tool
def get_customer_info(customer_id: str) -> str:
    """Get customer information including plan and history."""
    customer = db.get_customer(customer_id)
    return f"Name: {customer.name}, Plan: {customer.plan}, Since: {customer.created_at}"

@tool
def create_support_ticket(title: str, description: str, priority: str) -> str:
    """Create a support ticket for issues that need human attention."""
    ticket = ticketing.create(title=title, description=description, priority=priority)

    agentops.record(ActionEvent(
        action_type="ticket_created",
        params={"ticket_id": ticket.id, "priority": priority}
    ))

    return f"Ticket created: {ticket.id}"

# Create agent
tools = [search_knowledge_base, get_customer_info, create_support_ticket]
prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful customer support agent."),
    ("human", "{input}"),
    MessagesPlaceholder(variable_name="agent_scratchpad")
])

llm = ChatOpenAI(model="gpt-4", temperature=0)
agent = create_openai_functions_agent(llm, tools, prompt)
executor = AgentExecutor(agent=agent, tools=tools, verbose=True)

# Run with session tracking
def handle_customer_query(customer_id: str, query: str):
    with agentops.start_session(tags=[f"customer-{customer_id}"]):
        try:
            result = executor.invoke({
                "input": f"Customer ID: {customer_id}\nQuery: {query}"
            })

            agentops.record(ActionEvent(
                action_type="query_resolved",
                params={
                    "customer_id": customer_id,
                    "resolution_type": "automated"
                }
            ))

            agentops.end_session(end_state="Success")
            return result["output"]

        except Exception as e:
            agentops.end_session(end_state="Fail", end_state_reason=str(e))
            raise
