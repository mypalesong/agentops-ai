CrewAI Integration

AgentOps provides full observability for CrewAI, tracking crews, agents, tasks, and all inter-agent communication.

Quick Start​

import agentops
from crewai import Agent, Task, Crew

# Initialize AgentOps
agentops.init(api_key="your-agentops-api-key")

# Create agents
researcher = Agent(
    role="Researcher",
    goal="Research topics thoroughly",
    backstory="Expert researcher with attention to detail"
)

writer = Agent(
    role="Writer",
    goal="Write compelling content",
    backstory="Professional writer"
)

# Create tasks
research_task = Task(
    description="Research AI trends in 2024",
    agent=researcher,
    expected_output="Research summary"
)

write_task = Task(
    description="Write an article based on research",
    agent=writer,
    expected_output="Article draft"
)

# Create and run crew
crew = Crew(
    agents=[researcher, writer],
    tasks=[research_task, write_task]
)

result = crew.kickoff()
agentops.end_session(end_state="Success")

Agent Configuration​

Basic Agent​

agent = Agent(
    role="Data Analyst",
    goal="Analyze data and provide insights",
    backstory="Expert data analyst with statistics background",
    verbose=True
)

Agent with Tools​

from crewai_tools import SerperDevTool, WebsiteSearchTool

search_tool = SerperDevTool()
web_tool = WebsiteSearchTool()

researcher = Agent(
    role="Researcher",
    goal="Find accurate information",
    backstory="Skilled researcher",
    tools=[search_tool, web_tool],
    verbose=True
)

Agent with Memory​

agent = Agent(
    role="Assistant",
    goal="Help users effectively",
    backstory="Helpful AI assistant",
    memory=True,  # Enable memory
    verbose=True
)

Task Configuration​

Basic Task​

task = Task(
    description="Analyze the quarterly sales data",
    agent=analyst,
    expected_output="Sales analysis report"
)

Task with Context​

research_task = Task(
    description="Research market trends",
    agent=researcher,
    expected_output="Market research report"
)

analysis_task = Task(
    description="Analyze the research findings",
    agent=analyst,
    context=[research_task],  # Uses output from research_task
    expected_output="Analysis report"
)

Crew Configuration​

Sequential Crew​

crew = Crew(
    agents=[researcher, writer, editor],
    tasks=[research_task, write_task, edit_task],
    verbose=True,
    process=Process.sequential  # Tasks run in order
)

Hierarchical Crew​

from crewai import Process

manager = Agent(
    role="Project Manager",
    goal="Coordinate the team",
    backstory="Experienced project manager"
)

crew = Crew(
    agents=[researcher, writer, editor],
    tasks=[research_task, write_task, edit_task],
    manager_agent=manager,
    process=Process.hierarchical
)

Advanced Usage​

Multiple Crews​

agentops.init(api_key="your-agentops-api-key", auto_start_session=False)

projects = ["Project A", "Project B", "Project C"]

for project in projects:
    session = agentops.start_session(tags=["crewai", project])

    crew = Crew(
        agents=[...],
        tasks=[Task(description=f"Work on {project}", ...)]
    )

    result = crew.kickoff()
    session.end(end_state="Success")

Custom Callbacks​

def task_callback(task_output):
    """Called when a task completes."""
    agentops.record(Event(
        event_type="task_complete",
        params={
            "task": task_output.description,
            "output_length": len(task_output.raw)
        }
    ))

task = Task(
    description="Complete analysis",
    agent=analyst,
    callback=task_callback
)

Best Practices​

1. Tag Sessions by Crew​

session = agentops.start_session(tags=[
    "crewai",
    "research-crew",
    "production"
])

2. Enable Verbose Mode​

crew = Crew(
    agents=[...],
    tasks=[...],
    verbose=True  # Better debugging
)

3. Use Memory for Complex Tasks​

agent = Agent(
    role="Assistant",
    memory=True  # Remembers context
)
