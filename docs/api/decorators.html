Decorators

Decorators for tracking agents and tools.

@track_agent​

Track an agent class.

@track_agent(name: str = None, agent_id: str = None)

Parameters​

ParameterTypeDefaultDescription
namestrClass nameDisplay name for agent
agent_idstrAuto-generatedCustom agent ID

Example​

from agentops import track_agent

@track_agent(name="CustomerSupport")
class CustomerSupportAgent:
    def __init__(self):
        pass

    def handle_query(self, query):

@tool​

Track a tool/function.

@tool(name: str = None)

Parameters​

ParameterTypeDefaultDescription
namestrFunction nameDisplay name for tool

Example​

from agentops import tool

@tool(name="web_search")
def search_web(query: str) -> list:
    """Search the web for information."""
    results = perform_search(query)
    return results

@tool(name="calculator")
def calculate(expression: str) -> float:
    """Evaluate math expression."""
    return eval(expression)

Async Support​

@tool(name="async_fetch")
async def fetch_data(url: str) -> dict:
    """Fetch data asynchronously."""
    async with aiohttp.ClientSession() as session:
        async with session.get(url) as response:
            return await response.json()

Complete Example​

import agentops
from agentops import track_agent, tool

agentops.init(api_key="your-api-key")

@tool(name="search")
def search(query: str) -> list:
    return ["result1", "result2"]

@track_agent(name="ResearchAgent")
class ResearchAgent:
    def research(self, topic):
        results = search(topic)  # Tool call tracked
        return results

# Usage
agent = ResearchAgent()
agent.research("AI trends")

agentops.end_session(end_state="Success")
